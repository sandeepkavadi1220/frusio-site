<script>
  showCategory('all'); // Show all products initially

  function showCategory(category) {
    const products = document.querySelectorAll('.product');
    products.forEach(product => {
      if (category === 'all' || product.dataset.category === category) {
        product.style.display = 'block';
      } else {
        product.style.display = 'none';
      }
    });
    hideOrderForm();
  }

  const orderFormContainer = document.getElementById('orderFormContainer');
  const productNameInput = document.getElementById('productName');
  const priceInput = document.getElementById('price');
  const quantityInput = document.getElementById('quantity');
  const totalInput = document.getElementById('total');

  function showOrderForm(productName, price) {
    orderFormContainer.style.display = 'block';
    productNameInput.value = productName;
    priceInput.value = price;
    quantityInput.value = 1;
    totalInput.value = price;
    scrollTo(orderFormContainer.offsetTop - 20);
  }
  function hideOrderForm() {
    orderFormContainer.style.display = 'none';
  }
  function updateTotal() {
    let qty = parseInt(quantityInput.value);
    if (isNaN(qty) || qty < 1) qty = 1;
    quantityInput.value = qty;
    let total = qty * parseFloat(priceInput.value);
    totalInput.value = total.toFixed(2);
  }

  function submitOrder(e) {
    e.preventDefault();
    const paymentMethod = document.getElementById('paymentMethod').value;
    const deliveryPlace = document.getElementById('deliveryPlace').value;
    const address = document.getElementById('address').value.trim();

    if (!paymentMethod || !deliveryPlace || !address) {
      alert('Please fill all order details.');
      return;
    }

    alert(`Thank you for your order!
Product: ${productNameInput.value}
Quantity: ${quantityInput.value}
Total: ₹${totalInput.value}
Payment Method: ${paymentMethod}
Delivery Place: ${deliveryPlace}
Address: ${address}
We will contact you shortly.`);

    // Reset and hide form after submission
    e.target.reset();
    hideOrderForm();
  }
</script>
